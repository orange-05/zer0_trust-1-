# requirements.txt — Python Dependencies
# ========================================
# WHY PIN EXACT VERSIONS?
# - Unpinned deps can break your build when a new version releases.
# - "flask>=3.0" could install Flask 4.0 tomorrow with breaking changes.
# - Pinned versions ensure reproducible builds — same deps every time.
# - This is especially critical for security tools (audit specific versions).

# ── Core Flask Framework ──────────────────────────────────────────
Flask==3.1.0

# ── CORS — allows React frontend to call the API from browser ─────
Flask-Cors==6.0.0

# ── JWT Authentication ────────────────────────────────────────────
# Flask-JWT-Extended adds @jwt_required() decorator and token creation
Flask-JWT-Extended==4.7.1

# ── Password Hashing ─────────────────────────────────────────────
# Werkzeug is already a Flask dependency, listed for clarity
